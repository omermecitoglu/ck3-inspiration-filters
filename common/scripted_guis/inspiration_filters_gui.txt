inspiration_filters_toggle_level_1_inspirations = {
	scope = character
	is_shown = {
		NOT = {
			has_global_variable = ignore_level_1_inspirations
		}
	}
	effect = {
		if = {
			limit = {
				has_global_variable = ignore_level_1_inspirations
			}
			remove_global_variable = ignore_level_1_inspirations
		}
		else = {
			set_global_variable = ignore_level_1_inspirations
		}
	}
}
